\newcommand{\timeStep}
{
$
\gbox{~~~T_\text{old}~~}
-\frac{\gbox{~~~\delta t~~~}}{2\tims\gbox{~~~d~~~}}
\biggl(\gbox{~~\text{u}~~} ~\bigl( \gbox{~~~T_\text{west}~~}
                          - \gbox{~~~T_\text{east}~~} \bigr)
+      \gbox{~~\text{v}~~} ~\bigl( \gbox{~~~T_\text{north}~~}
                          - \gbox{~~~T_\text{south}~~} \bigr)
\biggr)
= \gbox{~~~T_\text{new}~~}$
}

\begin{slide}{Sunshine!}

\begin{list0}
\item The sun has come out in the south. 
    \begin{list1}
    \item Add 2\de C to the temperature in Swansea, Bristol and Southend
    \item Add 5\de C to the temperature in London due to the urban heat island
    \end{list1}
\end{list0}

\vskip 2ex
\heading{Mountains!}

\begin{list0}
\item The air is rising over the mountains. 
    \begin{list1}
    \item Subtract 4\de C from the temperature in Snowdon
    \end{list1}
\end{list0}

\end{slide}

\begin{slide}{Clear skies at night!}

\begin{list0}
\item The sun has set and clear skies lead to radiative cooling to space
    \begin{list1}
    \item Subtract 5\de C from the temperature in Peterborough and Norwich
    \end{list1}
\end{list0}

\end{slide}
